    // 자바스크립트가 잴 마지막에 실행되도록
    window.onload = function(){
        let currentIndex = 0; // 현재 이미지
        const slider = document.querySelectorAll(".slider"); // 모든 이미지 변수 저장
        
        // 이미지 변경
        setInterval(() => { // 3초에 한번씩 실행
            let nextIndex = (currentIndex + 1) % slider.length ;
            // 첫번째 이미지가 사라짐
            slider[currentIndex].style.opacity = "0";
            slider[nextIndex].style.opacity = "1";  
            slider.forEach(img => img.style.transition = "all 3s");
            
            currentIndex = nextIndex; // 두번 째 인덱스 값을 현재 인덱스 값에 저장       
        }, 3000)
    }


    $(function(){
        let currentIndex = 0; // 현재 이미지 
           setInterval(function(){ // 3s 한 번씩
            let nextIndex = (currentIndex +1) % 5;
               $(".slider").eq(currentIndex).fadeOut(1200);
            $(".slider").eq(nextIndex).fadeIn(1200);
               currentIndex = nextIndex;
        }, 3000)
    });
